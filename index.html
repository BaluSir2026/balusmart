üöÄ Quick Setup Guide - Google Sheets Integration

1 Create Google Sheet

1. Go to Google Sheets and create a new spreadsheet

2. Name it something like "Quiz Results"

3. Add these column headers in Row 1:

Timestamp | Name | Email | Student ID | Batch | Score | Total Questions | Percentage | Status | Time Taken

2 Create Google Apps Script

1. In your Google Sheet, click Extensions ‚Üí Apps Script

2. Delete any existing code

3. Copy and paste this code:

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  
  sheet.appendRow([
    new Date(),
    data.name,
    data.email,
    data.studentId,
    data.batch,
    data.score,
    data.totalQuestions,
    data.percentage,
    data.status,
    data.timeTaken
  ]);
  
  return ContentService.createTextOutput(JSON.stringify({
    'status': 'success',
    'message': 'Data saved successfully'
  })).setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  return ContentService.createTextOutput("Quiz API is working!");
}
