‚ö†Ô∏è First Time Setup Required

Follow these steps to connect your quiz to Google Sheets. You only need to do this once!

1 Create Google Spreadsheet

1. Go to Google Sheets

2. Create new spreadsheet: "Quiz Platform"

3. Create two sheets (tabs):

  • Questions - stores all quiz questions
  • Results - stores student results

2 Setup Questions Sheet

In "Questions" sheet, add headers in Row 1:

Question | OptionA | OptionB | OptionC | OptionD | CorrectAnswer | QuizID

Add your questions (example):

What is 2+2? | 3 | 4 | 5 | 6 | B | QUIZ001

3 Setup Results Sheet

In "Results" sheet, add headers in Row 1:

Timestamp | Name | Email | StudentID | Batch | QuizID | Score | TotalQuestions | Percentage | Status | TimeTaken

4 Create Apps Script

1. Click Extensions ‚Üí Apps Script

2. Delete existing code

3. Copy this code:

function doGet(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet();
  
  if (e.parameter.action === 'getQuestions') {
    var quizId = e.parameter.quizId || 'QUIZ001';
    var questionsSheet = sheet.getSheetByName('Questions');
    var data = questionsSheet.getDataRange().getValues();
    
    var questions = [];
    for (var i = 1; i < data.length; i++) {
      if (data[i][6] === quizId) {
        questions.push({
          question: data[i][0],
          options: [data[i][1], data[i][2], data[i][3], data[i][4]],
          correct: data[i][5]
        });
      }
    }
    
    return ContentService.createTextOutput(JSON.stringify({
      status: 'success',
      questions: questions
    })).setMimeType(ContentService.MimeType.JSON);
  }
  
  return ContentService.createTextOutput('Quiz API Active');
}

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet();
  var resultsSheet = sheet.getSheetByName('Results');
  var data = JSON.parse(e.postData.contents);
  
  resultsSheet.appendRow([
    new Date(),
    data.name,
    data.email,
    data.studentId,
    data.batch,
    data.quizId,
    data.score,
    data.totalQuestions,
    data.percentage,
    data.status,
    data.timeTaken
  ]);
  
  return ContentService.createTextOutput(JSON.stringify({
    status: 'success'
  })).setMimeType(ContentService.MimeType.JSON);
}
